# how to use :
#
# a. drag & drop aaa.csv and bbb.xlsx and ccc.map onto kintai-v3.bat at the same time.
#
# b. command line
# b.1. $ kintai-v3.bat --csv aaa.csv --xlsx bbb.xlsx --map ccc.map
# b.2. $ kintai-v3.bat --csv aaa.csv --xlsx bbb.xlsx --map ccc.map --computer ddd
#
# output (result.txt) format :
#
# key (= computer-name), YYYY-MM-DD, 1st hh:mm:ss, 2nd hh:mm:ss, 3rd hh:mm:ss, 4th hh:mm:ss, diff
#
# 1. 1st hh:mm:ss : duration from the first login to the last logoff
# 2. 2nd hh:mm:ss : net working time (sum of login sessions)
# 3. 3rd hh:mm:ss : total time spent logged off between sessions (1st hh:mm:ss - 2nd hh:mm:ss)
# 4. 4th hh:mm:ss : total working hours recorded in the Excel file
# 5. diff : difference between 4th hh:mm:ss and 2nd hh:mm:ss, which may indicate misrecorded time

SPECS = {
    'file' : [
        {
            'type' : 'csv',
            'required' : True,
            'encoding': 'cp932',
            'newline': ''
        },
        {
            'type' : 'xlsx',
            'required' : True,
            'encoding': None,
            'newline': None
        },
        {
            'type' : 'map',
            'required' : True,
            'encoding': 'utf-8',
            'newline': None
        }
    ],
    'data' : [
        {
            'name' : 'computer',
            'required' : False
        }
    ]
}

import sys
import csv
import datetime
import collections
import pathlib

def message(in_prefix, in_exit, *in_args):
    for arg in in_args:
        print(f'{in_prefix} : {arg}')
    if in_exit:
        sys.exit(1)

def error(*in_args):
    message('error', True, *in_args)

def warn(*in_args):
    message('warn', False, *in_args)

def required_value(in_dict, in_source, in_key):
    value = in_dict.get(in_key)
    if value is None:
        error(f'"{in_key}" is not found in {in_source}.')
    return value

def required_column(in_dict, in_column):
    return required_value(in_dict, 'csv/excel', in_column)

try:
    import openpyxl
except ImportError:
    error('exec "pip install openpyxl" at first.')

INPUT_FILE = {}
INPUT_PARM = {}

def generate_readme(in_specs):
    import string
    alphabet = list(string.ascii_lowercase)
    def generate_name():
        repeat = 3
        shift = alphabet.pop(0)
        return shift * repeat
    command = pathlib.Path(sys.argv[0]).name
    file_required = []
    parm_required = []
    parm_optional = []
    for spec in in_specs['file']: 
        filename = f"{generate_name()}.{spec['type']}"
        parm = f"--{spec['type']} {filename}"
        if spec['required']:
            file_required.append(filename)
            parm_required.append(parm)
        else:
            parm_optional.append(parm)
    for spec in in_specs['data']:
        parm = f"--{spec['name']} {generate_name()}"
        if spec['required']:
            parm_required.append(parm)
        else:
            parm_optional.append(parm)
    lines = []
    lines.append(f"# how to use :")
    lines.append(f"#")
    lines.append(f"# a. drag & drop {' and '.join(file_required)} onto {command} at the same time.")
    lines.append(f"#")
    lines.append(f"# b. command line")
    lines.append(f"# b.1. $ {command} {' '.join(parm_required)}")
    if parm_optional:
        lines.append(f"# b.2. $ {command} {' '.join(parm_required)} {' '.join(parm_optional)}")
    return "\n".join(lines)

if any(a.startswith('--') for a in sys.argv):
    import argparse
    parser = argparse.ArgumentParser()
    for spec in SPECS['file']:
        parser.add_argument(f"--{spec['type']}", required=spec['required'])
    for spec in SPECS['data']:
        parser.add_argument(f"--{spec['name']}", required=spec['required'])
    parsed = parser.parse_args()
    for spec in SPECS['file']:
        INPUT_FILE[spec['type']] = pathlib.Path(getattr(parsed, spec['type']))
    for spec in SPECS['data']:
        INPUT_PARM[spec['name']] = getattr(parsed, spec['name'])
else:
    required_cnt = sum(1 for spec in SPECS['file'] if spec['required'])
    if len(sys.argv) != required_cnt + 1:
        error(f'drag & drop {required_cnt} files')
    filenames = [pathlib.Path(a) for a in sys.argv[1:]]
    for spec in SPECS['file']:
        found = None
        for filename in filenames:
            if filename.suffix.lower() == f".{spec['type']}":
                found = filename
                break
        if found:
            INPUT_FILE[spec['type']] = found
        elif spec['required']:
            error(f"required type ({spec['type']}) is not found.")

def open_file(in_filename):
    for spec in SPECS['file']:
        filename = INPUT_FILE.get(spec['type'])
        if filename == in_filename:
            try:
                return open(in_filename, encoding=spec.get('encoding'), newline=spec.get('newline'))
            except Exception as err:
                error(f'failed to open {in_filename}', str(err))
    error(f'input file is not registered : {in_filename}')

CSV_COL_KEY = 'コンピューター名'
CSV_COL_LOGIN = 'ログイン名'
CSV_COL_OPERATION = '操作種別'
CSV_COL_OPERATION_S = '操作開始'
CSV_COL_OPERATION_E = '操作終了'
CSV_COL_TIME = '日時'

XLS_COL_KEY = '従業員番号'
XLS_COL_TIME = '日付'
XLS_COL_ACTUAL = '総労働時間'

# Workday starts at this hour (e.g., 4 means 04:00 to next day 03:59:59)
START_HOUR = 4

def extract_csv(in_filename, in_target):
    result = collections.defaultdict(list)
    with open_file(in_filename) as fh:
        reader = csv.DictReader(fh)
        fmt = '%Y/%m/%d %H:%M'
        for row in reader:
            op = required_column(row, CSV_COL_OPERATION)
            if op not in (CSV_COL_OPERATION_S, CSV_COL_OPERATION_E):
                continue
            if in_target and required_column(row, CSV_COL_KEY) != in_target:
                continue
            # dt = datetime.datetime.strptime(required_column(row, CSV_COL_TIME), fmt)
            try:
                dt = datetime.datetime.strptime(required_column(row, CSV_COL_TIME), fmt)
            except ValueError:
                error(f'{required_column(row, CSV_COL_TIME)} is not {fmt}.')
            if dt.hour < START_HOUR:
                work_day = (dt - datetime.timedelta(days=1)).date()
            else:
                work_day = dt.date()
            result[(
                required_column(row, CSV_COL_LOGIN),
                required_column(row, CSV_COL_KEY)
            )].append((dt, op, work_day))
    return result

CSV_ARR = extract_csv(INPUT_FILE['csv'], INPUT_PARM.get('computer'))

def load_mapping(in_filename):
    result = {}
    with open_file(in_filename) as fh:
        for line in fh:
            line = line.strip()
            if not line or line.startswith('#'):
                continue
            if ':' not in line:
                error(f'keys should be separated using ":" in {in_filename} ({line})')
            left, right = line.split(':', 1)
            result[left] = right
    return result

KEY_MAP = load_mapping(INPUT_FILE['map'])

def create_xlsx_map(in_filename):
    def string_to_date(in_str):
        if isinstance(in_str, datetime.datetime):
            return in_str.date()
        if isinstance(in_str, datetime.date):
            return in_str
        if isinstance(in_str, str):
            return datetime.datetime.strptime(in_str, '%Y/%m/%d').date()
        raise TypeError(f'unexpected date type: {type(in_str)}')
    result = collections.defaultdict(lambda: collections.defaultdict(datetime.timedelta))
    try:
        book = openpyxl.load_workbook(in_filename, read_only=True)
    except Exception as err:
        error(f'failed to open xlsx : {in_filename}', str(err))
    sheet = book.active
    header = None
    key_matched = 0
    for row in sheet.iter_rows(values_only=True):
        if not header:
            header = list(row)
            continue
        data = dict(zip(header, row))
        join_key = KEY_MAP.get(str(required_column(data, XLS_COL_KEY)))
        if join_key is None:
            continue
        else:
            key_matched += 1
        dt_raw = required_column(data, XLS_COL_TIME)
        actual_val = required_column(data, XLS_COL_ACTUAL)
        try:
            work_day = string_to_date(dt_raw)
        except Exception as err:
            warn(f'failed to parse date of {join_key}', str(err))
            continue
        if isinstance(actual_val, datetime.time):
            actual_val = datetime.timedelta(
                hours=actual_val.hour,
                minutes=actual_val.minute,
                seconds=actual_val.second
            )
        else:
            error(f'unexpected type for actual_val : {type(actual_val)}')
        result[join_key][work_day] = actual_val
    if key_matched == 0:
        warn(f'0 matched. mapping file may be wrong')
    book.close() 
    return result

XLSX_MAP = create_xlsx_map(INPUT_FILE['xlsx'])

def hours(td):
    return td.total_seconds() / 3600

for (login, match_key), entries in CSV_ARR.items():
    entries.sort()
    days = collections.defaultdict(list)
    for dt, op, work_day in entries:
        days[work_day].append((dt, op))
    for day, day_logs in sorted(days.items()):
        day_start = datetime.datetime.combine(day, datetime.time(START_HOUR, 0, 0))
        day_end = day_start + datetime.timedelta(days=1) - datetime.timedelta(seconds=1)
        start_time = None
        end_time = None
        sessions = []
        current_start = None
        for dt, op in day_logs:
            if dt < day_start or dt > day_end:
                continue
            if op == CSV_COL_OPERATION_S:
                if start_time is None:
                    start_time = dt
                current_start = dt
            elif op == CSV_COL_OPERATION_E:
                end_time = dt
                if current_start:
                    sessions.append((current_start, dt))
                    current_start = None
        if start_time is None:
            continue
        if end_time is None:
            end_time = day_end
        total_span = end_time - start_time
        actual = datetime.timedelta()
        for s, e in sessions:
            if e < s:
                continue
            actual += (e - s)
        diff = total_span - actual
        if diff < datetime.timedelta():
            diff = datetime.timedelta()
        excel_actual = XLSX_MAP.get(match_key, {}).get(day)
        if excel_actual is not None:
            excel_diff_val = excel_actual - actual 
            res_excel_actual = str(excel_actual)
            res_diff = f"{hours(excel_diff_val):.2f}"
        else:
            res_excel_actual = "NO_EXCEL_DATA"
            res_diff = "-"
        print(f'{login} ({match_key}), {day}, {total_span}, {actual}, {diff}, {res_excel_actual}, {res_diff}')

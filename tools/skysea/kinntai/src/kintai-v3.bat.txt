# how to use :
#
# a. drag & drop aaa.csv and bbb.xlsx and ccc.map onto kintai-v3.bat at the same time.
#
# b. command line
# b.1. $ kintai-v3.bat --csv aaa.csv --xlsx bbb.xlsx --map ccc.map
# b.2. $ kintai-v3.bat --csv aaa.csv --xlsx bbb.xlsx --map ccc.map > output.txt
# b.3. $ kintai-v3.bat --csv aaa.csv --xlsx bbb.xlsx --map ccc.map --computer ddd
# b.4. $ kintai-v3.bat --csv aaa.csv --xlsx bbb.xlsx --map ccc.map --computer ddd > output.txt
#
# output format :
#
# key (= computer-name), YYYY-MM-DD, 1st hh:mm:ss, 2nd hh:mm:ss, 3rd hh:mm:ss, 4th hh:mm:ss, diff
#
# 1. 1st hh:mm:ss : duration from the first login to the last logoff
# 2. 2nd hh:mm:ss : net working time (sum of login sessions)
# 3. 3rd hh:mm:ss : total time spent logged off between sessions (1st hh:mm:ss - 2nd hh:mm:ss)
# 4. 4th hh:mm:ss : total working hours recorded in the Excel file
# 5. diff : difference between 4th hh:mm:ss and 2nd hh:mm:ss, which may indicate misrecorded time

    'file' : [
        {

            'required' : False

INPUT_FILE = {}

def generate_readme(in_specs):
    import string
    alphabet = list(string.ascii_lowercase)
    def generate_name():
        repeat = 3
        shift = alphabet.pop(0)
        return shift * repeat
    command = pathlib.Path(sys.argv[0]).name
    file_required = []
    parm_required = []
    parm_optional = []
    for spec in in_specs['file']: 
        filename = f"{generate_name()}.{spec['type']}"
        parm = f"--{spec['type']} {filename}"
        if spec['required']:
            file_required.append(filename)
            parm_required.append(parm)
        else:
            parm_optional.append(parm)
    for spec in in_specs['data']:
        parm = f"--{spec['name']} {generate_name()}"
        if spec['required']:
            parm_required.append(parm)
        else:
            parm_optional.append(parm)
    lines = []
    lines.append(f"# how to use :")
    lines.append(f"#")
    lines.append(f"# a. drag & drop {' and '.join(file_required)} onto {command} at the same time.")
    lines.append(f"#")
    lines.append(f"# b. command line")
    lines.append(f"# b.1. $ {command} {' '.join(parm_required)}")
    lines.append(f"# b.2. $ {command} {' '.join(parm_required)} > output.txt")
    lines.append(f"# b.3. $ {command} {' '.join(parm_required)} {' '.join(parm_optional)}")
    lines.append(f"# b.4. $ {command} {' '.join(parm_required)} {' '.join(parm_optional)} > output.txt")
    return "\n".join(lines)

ENCODING = 'cp932'

CSV_COL_KEY = 'コンピューター名'
CSV_COL_LOGIN = 'ログイン名'
CSV_COL_OPERATION = '操作種別'
CSV_COL_OPERATION_S = '操作開始'
CSV_COL_OPERATION_E = '操作終了'
CSV_COL_TIME = '日時'

XLS_COL_KEY = '従業員番号'
XLS_COL_TIME = '日付'
XLS_COL_ACTUAL = '総労働時間'

# Workday starts at this hour (e.g., 4 means 04:00 to next day 03:59:59)
START_HOUR = 4

def extract_csv(in_filename, in_target):
    result = collections.defaultdict(list)
    with open(in_filename, encoding=ENCODING, newline='') as fh:
        reader = csv.DictReader(fh)
        for row in reader:
            op = row[CSV_COL_OPERATION]
            if op not in (CSV_COL_OPERATION_S, CSV_COL_OPERATION_E):
                continue
            if in_target and row[CSV_COL_KEY] != in_target:
                continue
            dt = datetime.datetime.strptime(row[CSV_COL_TIME], '%Y/%m/%d %H:%M')
            if dt.hour < START_HOUR:
                work_day = (dt - datetime.timedelta(days=1)).date()
            else:
                work_day = dt.date()
            result[(row[CSV_COL_LOGIN], row[CSV_COL_KEY])].append((dt, op, work_day))
    return result

CSV_ARR = extract_csv(INPUT_FILE['csv'], INPUT_DATA.get('computer'))

def load_mapping(filename):
    result = {}
    with open(filename, encoding='utf-8') as fh:
        for line in fh:
            line = line.strip()
            if not line or line.startswith('#'):
                continue
            left, right = line.split(':', 1)
            result[left] = right
    return result

KEY_MAP = load_mapping(INPUT_FILE['map'])

def create_xlsx_map(in_filename):
    def string_to_date(in_str):
        if isinstance(in_str, datetime.datetime):
            return in_str.date()
        if isinstance(in_str, datetime.date):
            return in_str
        if isinstance(in_str, str):
            return datetime.datetime.strptime(in_str, '%Y/%m/%d').date()
        raise TypeError(f'unexpected date type: {type(in_str)}')
    result = collections.defaultdict(lambda: collections.defaultdict(datetime.timedelta))
    book = openpyxl.load_workbook(in_filename, read_only=True)  
    sheet = book.active
    header = None
    for row in sheet.iter_rows(values_only=True):
        if not header:
            header = list(row)
            continue
        data = dict(zip(header, row))
        join_key = KEY_MAP.get(str(data.get(XLS_COL_KEY)), 'no-match')
        dt_raw = data.get(XLS_COL_TIME)
        actual_val = data.get(XLS_COL_ACTUAL)
        if join_key is None or dt_raw is None or actual_val is None:
            continue
        try:
            work_day = string_to_date(dt_raw)
        except Exception:
            continue
        if isinstance(actual_val, datetime.time):
            actual_val = datetime.timedelta(
                hours=actual_val.hour,
                minutes=actual_val.minute,
                seconds=actual_val.second
            )
        else:
            raise TypeError(f'unexpected type for actual_val: {type(actual_val)}')
        result[join_key][work_day] = actual_val
    book.close() 
    return result

XLSX_MAP = create_xlsx_map(INPUT_FILE['xlsx'])

def hours(td):
    return td.total_seconds() / 3600

for (login, match_key), entries in CSV_ARR.items():
    entries.sort()
    days = collections.defaultdict(list)
    for dt, op, work_day in entries:
        days[work_day].append((dt, op))
    for day, day_logs in sorted(days.items()):
        day_start = datetime.datetime.combine(day, datetime.time(START_HOUR, 0, 0))
        day_end = day_start + datetime.timedelta(days=1) - datetime.timedelta(seconds=1)
        start_time = None
        end_time = None
        sessions = []
        current_start = None
        for dt, op in day_logs:
            if dt < day_start or dt > day_end:
                continue
            if op == CSV_COL_OPERATION_S:
                if start_time is None:
                    start_time = dt
                current_start = dt
            elif op == CSV_COL_OPERATION_E:
                end_time = dt
                if current_start:
                    sessions.append((current_start, dt))
                    current_start = None
        if start_time is None:
            continue
        if end_time is None:
            end_time = day_end
        total_span = end_time - start_time
        actual = datetime.timedelta()
        for s, e in sessions:
            if e < s:
                continue
            actual += (e - s)
        diff = total_span - actual
        if diff < datetime.timedelta():
            diff = datetime.timedelta()
        excel_actual = XLSX_MAP.get(match_key, {}).get(day)
        if excel_actual is not None:
            excel_diff_val = excel_actual - actual 
            res_excel_actual = str(excel_actual)
            res_diff = f"{hours(excel_diff_val):.2f}"
        else:
            res_excel_actual = "NO_EXCEL_DATA"
            res_diff = "-"
        print(f'{login} ({match_key}), {day}, {total_span}, {actual}, {diff}, {res_excel_actual}, {res_diff}')
